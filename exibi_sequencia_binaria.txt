        if (n & 1 )
        {
            printf ( " 1 " );
        }
        outro
        {
            printf ( " 0 " );
        }
    }

        }

        }


    }
    printf ( " \ n " );

        nd = n >> i;
        if (n & 1 )
        {
            printf ( " 1 " );
        }
        outro
        {
            printf ( " 0 " );
        }


    }
